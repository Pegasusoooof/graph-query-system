### 1.2 Preprocessing Pipeline (Python)

```python
# preprocessing/pipeline.py
import pandas as pd
import uuid
from datetime import datetime
from supabase import create_client

class OTCPreprocessor:
    def __init__(self, supabase_url, supabase_key):
        self.db = create_client(supabase_url, supabase_key)
        self.id_map = {}  # source_id -> uuid, for FK resolution
    
    def normalize_id(self, entity_type: str, source_id: str) -> str:
        """Deterministic UUID from source ID — idempotent on re-runs."""
        key = f"{entity_type}:{source_id}"
        if key not in self.id_map:
            self.id_map[key] = str(uuid.uuid5(uuid.NAMESPACE_DNS, key))
        return self.id_map[key]
    
    def clean_dataframe(self, df: pd.DataFrame) -> pd.DataFrame:
        """Standard cleaning applied to all entities."""
        df = df.drop_duplicates()
        df.columns = [c.strip().lower().replace(" ", "_") for c in df.columns]
        # Coerce date columns
        for col in df.columns:
            if "date" in col or "time" in col:
                df[col] = pd.to_datetime(df[col], errors="coerce")
        # Strip whitespace from string columns
        str_cols = df.select_dtypes("object").columns
        df[str_cols] = df[str_cols].apply(lambda x: x.str.strip())
        return df
    
    def process_sales_orders(self, filepath: str):
        df = self.clean_dataframe(pd.read_csv(filepath))
        records = []
        for _, row in df.iterrows():
            records.append({
                "id": self.normalize_id("order", row["sales_order"]),
                "order_id": row["sales_order"],
                "customer_id": self.normalize_id("customer", row["sold_to_party"]),
                "order_date": row.get("order_date"),
                "status": row.get("order_status"),
                "net_value": row.get("net_value"),
                "currency": row.get("currency", "USD"),
            })
        self.db.table("sales_orders").upsert(records).execute()
    
    def validate_referential_integrity(self):
        """Detect and report broken FK links before graph construction."""
        issues = []
        result = self.db.rpc("check_orphaned_deliveries").execute()
        if result.data:
            issues.append(f"{len(result.data)} deliveries with no matching order")
        result = self.db.rpc("check_unbilled_deliveries").execute()
        if result.data:
            issues.append(f"{len(result.data)} deliveries never billed")
        return issues
```

## Phase 2: Graph Data Model

The graph has two node classes — **entity nodes** (the business objects) and **value nodes** (shared reference data like plants, GL accounts). Here's the node/edge taxonomy:### 2.1 Graph Construction (NetworkX)

```python
# graph/builder.py
import networkx as nx
import json
from supabase import create_client

class OTCGraphBuilder:
    def __init__(self, supabase_url, supabase_key):
        self.db = create_client(supabase_url, supabase_key)
        self.G = nx.DiGraph()

    def build(self) -> nx.DiGraph:
        self._add_customers()
        self._add_products()
        self._add_sales_orders()
        self._add_deliveries()
        self._add_billing()
        self._add_journal_entries()
        self._add_edges()
        return self.G

    def _add_customers(self):
        rows = self.db.table("customers").select("*").execute().data
        for r in rows:
            self.G.add_node(
                r["id"],
                entity_type="Customer",
                label=r.get("name", r["customer_id"]),
                customer_id=r["customer_id"],
                region=r.get("region"),
                country=r.get("country"),
            )

    def _add_sales_orders(self):
        rows = self.db.table("sales_orders").select("*").execute().data
        for r in rows:
            self.G.add_node(
                r["id"],
                entity_type="SalesOrder",
                label=f"SO-{r['order_id']}",
                order_id=r["order_id"],
                order_date=str(r.get("order_date")),
                net_value=float(r.get("net_value") or 0),
                status=r.get("status"),
                currency=r.get("currency"),
            )

    def _add_billing(self):
        rows = self.db.table("billing_documents").select("*").execute().data
        for r in rows:
            self.G.add_node(
                r["id"],
                entity_type="BillingDocument",
                label=f"BILL-{r['billing_id']}",
                billing_id=r["billing_id"],
                billing_date=str(r.get("billing_date")),
                net_value=float(r.get("net_value") or 0),
                document_type=r.get("document_type"),
            )

    def _add_edges(self):
        # Customer -> Sales Order
        for r in self.db.table("sales_orders").select("id, customer_id").execute().data:
            if r["customer_id"]:
                self.G.add_edge(r["customer_id"], r["id"],
                    relationship="PLACES_ORDER", weight=1)

        # Sales Order -> Delivery
        for r in self.db.table("deliveries").select("id, order_id").execute().data:
            if r["order_id"]:
                self.G.add_edge(r["order_id"], r["id"],
                    relationship="FULFILLED_BY", weight=1)

        # Delivery -> Billing
        for r in self.db.table("billing_documents").select("id, delivery_id").execute().data:
            if r["delivery_id"]:
                self.G.add_edge(r["delivery_id"], r["id"],
                    relationship="BILLED_AS", weight=1)

        # Billing -> Journal Entry
        for r in self.db.table("journal_entries").select("id, billing_id").execute().data:
            if r["billing_id"]:
                self.G.add_edge(r["billing_id"], r["id"],
                    relationship="POSTS_TO_JOURNAL", weight=1)

    def to_json(self) -> dict:
        """Serialize for react-force-graph."""
        nodes = []
        links = []
        for node_id, attrs in self.G.nodes(data=True):
            nodes.append({"id": node_id, **attrs})
        for src, tgt, attrs in self.G.edges(data=True):
            links.append({"source": src, "target": tgt, **attrs})
        return {"nodes": nodes, "links": links}

    def detect_broken_flows(self) -> dict:
        """Identify incomplete O2C flows for query type (c)."""
        delivered_not_billed = []
        billed_without_delivery = []
        
        delivery_nodes = {n for n, d in self.G.nodes(data=True)
                         if d.get("entity_type") == "Delivery"}
        billing_nodes = {n for n, d in self.G.nodes(data=True)
                        if d.get("entity_type") == "BillingDocument"}
        
        # Deliveries with no outgoing edge to a BillingDocument
        for d in delivery_nodes:
            successors = list(self.G.successors(d))
            has_billing = any(
                self.G.nodes[s].get("entity_type") == "BillingDocument"
                for s in successors
            )
            if not has_billing:
                delivered_not_billed.append(self.G.nodes[d])
        
        # Billing documents with no delivery predecessor
        for b in billing_nodes:
            predecessors = list(self.G.predecessors(b))
            has_delivery = any(
                self.G.nodes[p].get("entity_type") == "Delivery"
                for p in predecessors
            )
            if not has_delivery:
                billed_without_delivery.append(self.G.nodes[b])
        
        return {
            "delivered_not_billed": delivered_not_billed,
            "billed_without_delivery": billed_without_delivery,
        }
```

```python
# api/main.py
from fastapi import FastAPI, HTTPException
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
import anthropic
import networkx as nx
import json

app = FastAPI()
client = anthropic.Anthropic()  # uses ANTHROPIC_API_KEY env var

# Load graph at startup
graph_data: dict = {}
G: nx.DiGraph = None

@app.on_event("startup")
async def startup():
    global G, graph_data
    builder = OTCGraphBuilder(SUPABASE_URL, SUPABASE_KEY)
    G = builder.build()
    graph_data = builder.to_json()

# ── Graph endpoints ───────────────────────────────────────────────
@app.get("/graph")
async def get_full_graph():
    return graph_data

@app.get("/graph/node/{node_id}")
async def get_node(node_id: str):
    if node_id not in G.nodes:
        raise HTTPException(404)
    attrs = dict(G.nodes[node_id])
    neighbors = [
        {
            "id": n,
            "relationship": G.edges[node_id, n].get("relationship"),
            **G.nodes[n],
        }
        for n in G.successors(node_id)
    ] + [
        {
            "id": n,
            "relationship": G.edges[n, node_id].get("relationship"),
            **G.nodes[n],
            "direction": "incoming",
        }
        for n in G.predecessors(node_id)
    ]
    return {"node": attrs, "connections": neighbors}

@app.get("/graph/trace/{billing_id}")
async def trace_flow(billing_id: str):
    """Full O2C trace: Sales Order → Delivery → Billing → Journal Entry."""
    billing_node = next(
        (n for n, d in G.nodes(data=True)
         if d.get("billing_id") == billing_id), None
    )
    if not billing_node:
        raise HTTPException(404, f"Billing doc {billing_id} not found")
    
    chain = {"billing": G.nodes[billing_node]}
    
    # Walk backwards to delivery, then to order
    for pred in G.predecessors(billing_node):
        if G.nodes[pred].get("entity_type") == "Delivery":
            chain["delivery"] = G.nodes[pred]
            for order_pred in G.predecessors(pred):
                if G.nodes[order_pred].get("entity_type") == "SalesOrder":
                    chain["sales_order"] = G.nodes[order_pred]
    
    # Walk forwards to journal entries
    chain["journal_entries"] = [
        G.nodes[s] for s in G.successors(billing_node)
        if G.nodes[s].get("entity_type") == "JournalEntry"
    ]
    return chain
```

### 3.2 Conversational Query Interface with Guardrails

This is the most critical component. The key design is: Claude does **not** answer from general knowledge — it only reasons over a structured context object built from the graph.

```python
# api/query.py
class QueryRequest(BaseModel):
    message: str
    conversation_history: list[dict] = []
    highlighted_nodes: list[str] = []  # currently selected nodes in UI

DOMAIN_SYSTEM_PROMPT = """
You are a data analysis assistant for an Order-to-Cash (O2C) business process dataset.

STRICT RULES:
1. You ONLY answer questions about the data in the provided context. 
2. If a question is unrelated to O2C business processes, orders, deliveries, invoices, 
   payments, customers, or products in this dataset, respond: 
   "I can only answer questions about the Order-to-Cash dataset. Please ask about 
   orders, deliveries, billing documents, journal entries, customers, or products."
3. NEVER invent, estimate, or hallucinate data. Every claim must be traceable to 
   the context provided.
4. When referencing a specific entity, always include its ID so the UI can highlight it.
5. Return node IDs to highlight in this format at the end of your response:
   <highlight_nodes>id1,id2,id3</highlight_nodes>

You are analyzing the following data context:
{context}
"""

def build_context_for_query(query: str, G: nx.DiGraph, db) -> str:
    """
    Smart context builder: pulls only relevant graph/SQL data for the query.
    Avoids stuffing the full graph into the prompt.
    """
    context_parts = []
    
    # Detect query intent
    query_lower = query.lower()
    
    if any(kw in query_lower for kw in ["billing document", "bill", "invoice"]):
        # Pull billing summary stats
        billing_nodes = [
            d for _, d in G.nodes(data=True)
            if d.get("entity_type") == "BillingDocument"
        ]
        context_parts.append(f"Total billing documents: {len(billing_nodes)}")
        context_parts.append("Sample billing documents: " + 
            json.dumps(billing_nodes[:5], default=str))
    
    if any(kw in query_lower for kw in ["broken", "incomplete", "missing", "not billed"]):
        broken = OTCGraphBuilder.detect_broken_flows_static(G)
        context_parts.append("Broken flows: " + json.dumps(broken, default=str))
    
    if any(kw in query_lower for kw in ["product", "material", "highest"]):
        # SQL query for product-billing count
        result = db.rpc("products_by_billing_count").execute()
        context_parts.append("Products by billing count: " + 
            json.dumps(result.data[:20], default=str))
    
    # Always include graph summary
    entity_counts = {}
    for _, d in G.nodes(data=True):
        etype = d.get("entity_type", "Unknown")
        entity_counts[etype] = entity_counts.get(etype, 0) + 1
    context_parts.append("Graph summary: " + json.dumps(entity_counts))
    
    return "\n\n".join(context_parts)

@app.post("/query")
async def natural_language_query(req: QueryRequest):
    context = build_context_for_query(req.message, G, db)
    system = DOMAIN_SYSTEM_PROMPT.format(context=context)
    
    messages = req.conversation_history + [{"role": "user", "content": req.message}]
    
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1500,
        system=system,
        messages=messages,
    )
    
    raw_text = response.content[0].text
    
    # Parse highlighted node IDs from response
    highlighted = []
    if "<highlight_nodes>" in raw_text:
        import re
        match = re.search(r"<highlight_nodes>(.*?)</highlight_nodes>", raw_text)
        if match:
            highlighted = match.group(1).split(",")
        raw_text = re.sub(r"<highlight_nodes>.*?</highlight_nodes>", "", raw_text).strip()
    
    return {
        "response": raw_text,
        "highlighted_nodes": highlighted,
        "model_used": "claude-sonnet-4-20250514",
    }

@app.post("/query/stream")
async def stream_query(req: QueryRequest):
    """Streaming version for real-time responses."""
    context = build_context_for_query(req.message, G, db)
    system = DOMAIN_SYSTEM_PROMPT.format(context=context)
    messages = req.conversation_history + [{"role": "user", "content": req.message}]
    
    def generate():
        with client.messages.stream(
            model="claude-sonnet-4-20250514",
            max_tokens=1500,
            system=system,
            messages=messages,
        ) as stream:
            for text in stream.text_stream:
                yield f"data: {json.dumps({'token': text})}\n\n"
        yield "data: [DONE]\n\n"
    
    return StreamingResponse(generate(), media_type="text/event-stream")
```

```tsx
"use client";
import { useCallback, useRef, useState } from "react";
import ForceGraph2D from "react-force-graph-2d";

const NODE_COLORS: Record<string, string> = {
  Customer:        "#1D9E75",  // teal
  SalesOrder:      "#378ADD",  // blue
  Delivery:        "#7F77DD",  // purple
  BillingDocument: "#D85A30",  // coral

export default function GraphView({
  graphData,
  highlightedNodes = [],
  onNodeClick,
}: {
  graphData: { nodes: any[]; links: any[] };
  highlightedNodes?: string[];
  onNodeClick: (node: any) => void;
}) {
  const fgRef = useRef<any>();

  const nodeCanvasObject = useCallback(
    (node: any, ctx: CanvasRenderingContext2D, globalScale: number) => {
      const isHighlighted = highlightedNodes.includes(node.id);
      const color = NODE_COLORS[node.entity_type] ?? "#888780";
      const radius = isHighlighted ? 8 : 5;

      // Glow effect for highlighted nodes
      if (isHighlighted) {
        ctx.shadowColor = color;
        ctx.shadowBlur = 12;
      }
      ctx.beginPath();
      ctx.arc(node.x, node.y, radius, 0, 2 * Math.PI);
      ctx.fillStyle = color;
      ctx.fill();
      ctx.shadowBlur = 0;

      // Label at higher zoom
      if (globalScale > 1.5) {
        ctx.font = `${10 / globalScale}px sans-serif`;
        ctx.fillStyle = "#fff";
        ctx.textAlign = "center";
        ctx.fillText(node.label ?? node.entity_type, node.x, node.y + radius + 6);
      }
    },
    [highlightedNodes]
  );

  return (
    <ForceGraph2D
      ref={fgRef}
      graphData={graphData}
      nodeCanvasObject={nodeCanvasObject}
      nodePointerAreaPaint={(node: any, color, ctx) => {
        ctx.beginPath();
        ctx.arc(node.x, node.y, 8, 0, 2 * Math.PI);
        ctx.fillStyle = color;
        ctx.fill();
      }}
      onNodeClick={onNodeClick}
      linkColor={() => "rgba(100,120,180,0.25)"}
      linkWidth={0.5}
      linkDirectionalArrowLength={3}
      linkDirectionalArrowRelPos={1}
      backgroundColor="transparent"
      cooldownTicks={200}
    />
  );
}
```

```tsx
// components/ChatPanel.tsx
"use client";
import { useState, useRef, useEffect } from "react";

export function ChatPanel({ onHighlight }: { onHighlight: (ids: string[]) => void }) {
  const [messages, setMessages] = useState<{role:string; content:string}[]>([
    { role: "assistant", content: "Hi! I can help you analyze the Order to Cash process." }
  ]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const bottomRef = useRef<HTMLDivElement>(null);

  const send = async () => {
    if (!input.trim() || loading) return;
    const userMsg = { role: "user", content: input };
    const history = [...messages, userMsg];
    setMessages(history);
    setInput("");
    setLoading(true);

    let assistantText = "";
    setMessages(prev => [...prev, { role: "assistant", content: "▌" }]);

    try {
      const res = await fetch("/api/query/stream", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          message: input,
          conversation_history: messages.map(m => ({
            role: m.role, content: m.content
          })),
        }),
      });

      const reader = res.body!.getReader();
      const decoder = new TextDecoder();

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        const chunk = decoder.decode(value);
        for (const line of chunk.split("\n")) {
          if (line.startsWith("data: ") && line !== "data: [DONE]") {
            const data = JSON.parse(line.slice(6));
            assistantText += data.token;
            setMessages(prev => [
              ...prev.slice(0, -1),
              { role: "assistant", content: assistantText + "▌" }
            ]);
          }
        }
      }

      // Final message — strip cursor, extract highlights
      const highlightMatch = assistantText.match(/<highlight_nodes>(.*?)<\/highlight_nodes>/);
      if (highlightMatch) {
        onHighlight(highlightMatch[1].split(",").map(s => s.trim()));
        assistantText = assistantText.replace(/<highlight_nodes>.*?<\/highlight_nodes>/g, "").trim();
      }
      setMessages(prev => [...prev.slice(0,-1), { role: "assistant", content: assistantText }]);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => { bottomRef.current?.scrollIntoView({ behavior: "smooth" }); }, [messages]);

  return (
    <div className="flex flex-col h-full">
      <div className="flex-1 overflow-y-auto p-4 space-y-3">
        {messages.map((m, i) => (
          <div key={i} className={`flex ${m.role === "user" ? "justify-end" : "justify-start"}`}>
            <div className={`max-w-[85%] rounded-xl px-3 py-2 text-sm whitespace-pre-wrap
              ${m.role === "user"
                ? "bg-gray-900 text-white rounded-br-sm"
                : "bg-gray-100 text-gray-800 rounded-bl-sm"}`}>
              {m.content}
            </div>
          </div>
        ))}
        <div ref={bottomRef}/>
      </div>
      <div className="p-3 border-t border-gray-100">
        <div className="flex gap-2">
          <input
            value={input}
            onChange={e => setInput(e.target.value)}
            onKeyDown={e => e.key === "Enter" && send()}
            placeholder="Analyze anything"
            className="flex-1 text-sm border border-gray-200 rounded-lg px-3 py-2 
                       focus:outline-none focus:ring-1 focus:ring-gray-400"
          />
          <button
            onClick={send}
            disabled={loading}
            className="px-4 py-2 bg-gray-900 text-white text-sm rounded-lg 
                       disabled:opacity-40 hover:bg-gray-700 transition-colors">
            Send
          </button>
        </div>
      </div>
    </div>
  );
}
```

## Phase 5: Design Considerations for Scale

Here are the key architectural decisions as you grow the system:

**Graph persistence strategy.** Start with in-memory NetworkX + pickle/JSON serialization. At ~1M nodes, migrate to Neo4j or RedisGraph. Keep the same Python interface with an adapter pattern so the API layer doesn't change.

**Context window management for queries.** The `build_context_for_query` function above is intentionally lean — it pulls targeted data rather than dumping the whole graph. As the dataset grows, add a vector embedding layer (pgvector in Supabase) on entity descriptions to do semantic retrieval before stuffing the LLM context.

**Broken flow detection as a precomputed view.** The `detect_broken_flows()` method traverses the graph on every call. In production, run this as a background job and store results in a Postgres materialized view that gets refreshed incrementally on data load.

**Conversation memory.** The current design threads the full `conversation_history` on each request, which works well up to ~10 turns. For longer sessions, summarize the earlier turns into a "session context" string and prepend it, keeping the sliding window to the last 5 messages.

**The guardrail approach.** The system prompt is the primary defense — but add a lightweight classifier as a pre-filter. Before calling Claude, check if the query contains O2C keywords (`order`, `delivery`, `billing`, `customer`, `invoice`, `journal`, `material`, `plant`, `payment`). If none match, reject immediately with a canned response. This avoids LLM costs on off-topic probes.
